FAQ

Is this a virus?

Nope, this is not a virus, in fact, you may check the original source code if you like:

(function($){
  $('video').each(function(){
    var el = $(this),
        parent = el.closest('.post-container'),
        button = $('<a href=\'#\' class=\'btn btn-le-gif\' target=\'_blank\'>Get Le Gif</a>');
    if (!$) {
        alert('No jQuery, can\'t continue');
        return;
    }
    if ( parent.next('.btn-le-gif').length ) return;
    button.css({ width: 458, marginTop: 10 });
    button.attr('href', el.attr('poster').replace(/s\.(jpg|gif|png)/i, 'sa.gif'));
    parent.after(button);
  });
})(jQuery);
